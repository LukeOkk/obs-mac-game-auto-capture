cmake_minimum_required(VERSION 3.28...3.30)

project(obs-mac-game-auto-capture VERSION 0.1.0)

if(NOT APPLE)
    message(FATAL_ERROR "This plugin is macOS-only.")
endif()

set(CMAKE_OSX_DEPLOYMENT_TARGET "14.0" CACHE STRING "macOS deployment target")
set(CMAKE_OSX_ARCHITECTURES "arm64" CACHE STRING "macOS architectures")
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

# libobs paths are passed in from build.sh (or set manually).
#   LIBOBS_INCLUDE_DIR -> /path/to/obs-studio/libobs (folder containing obs-module.h)
#   LIBOBS_LIBRARY     -> /Applications/OBS.app/Contents/Frameworks/libobs.framework/libobs
if(NOT LIBOBS_INCLUDE_DIR OR NOT EXISTS "${LIBOBS_INCLUDE_DIR}/obs-module.h")
    message(FATAL_ERROR
        "LIBOBS_INCLUDE_DIR not set or obs-module.h missing.\n"
        "Pass: -DLIBOBS_INCLUDE_DIR=/path/to/obs-studio/libobs")
endif()
if(NOT LIBOBS_LIBRARY OR NOT EXISTS "${LIBOBS_LIBRARY}")
    message(FATAL_ERROR
        "LIBOBS_LIBRARY not set or not found.\n"
        "Pass: -DLIBOBS_LIBRARY=/Applications/OBS.app/Contents/Frameworks/libobs.framework/libobs")
endif()

add_library(${PROJECT_NAME} MODULE)

target_sources(${PROJECT_NAME}
    PRIVATE
        src/plugin-main.c
        src/game-auto-source.m
)

# obsconfig.h is generated by the obs-studio build; we synthesize a minimal
# one here so the plugin can include obs-module.h without the full OBS build.
set(OBS_RELEASE_CANDIDATE 0)
set(OBS_BETA 0)
configure_file(
    "${LIBOBS_INCLUDE_DIR}/obsconfig.h.in"
    "${CMAKE_BINARY_DIR}/generated/obsconfig.h"
    @ONLY
)

target_include_directories(${PROJECT_NAME} PRIVATE
    "${CMAKE_BINARY_DIR}/generated"
    "${LIBOBS_INCLUDE_DIR}"
)

# obs-deps headers (simde, etc.) when provided
if(OBS_DEPS_INCLUDE_DIR)
    target_include_directories(${PROJECT_NAME} PRIVATE "${OBS_DEPS_INCLUDE_DIR}")
endif()

target_link_libraries(${PROJECT_NAME}
    PRIVATE
        "${LIBOBS_LIBRARY}"
        "-framework Foundation"
        "-framework AppKit"
        "-framework CoreGraphics"
        "-framework CoreMedia"
        "-framework CoreVideo"
        "-framework ScreenCaptureKit"
        "-framework IOSurface"
)

set_target_properties(${PROJECT_NAME} PROPERTIES
    BUNDLE TRUE
    BUNDLE_EXTENSION "plugin"
    MACOSX_BUNDLE TRUE
    MACOSX_BUNDLE_INFO_PLIST "${CMAKE_SOURCE_DIR}/src/Info.plist"
    OUTPUT_NAME "${PROJECT_NAME}"
    XCODE_ATTRIBUTE_PRODUCT_BUNDLE_IDENTIFIER "io.github.obs-plugins.mac-game-auto-capture"
)

target_compile_options(${PROJECT_NAME} PRIVATE
    -Wall
    -Wextra
    -fobjc-arc
)

# Note: actual install is handled by build.sh which copies the .plugin bundle
# to ~/Library/Application Support/obs-studio/plugins/ and injects data/ into
# Contents/Resources/. We deliberately do not define an install() rule here
# because cmake install(TARGETS ... LIBRARY) does not handle BUNDLE outputs
# correctly on macOS.
